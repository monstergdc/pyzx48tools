126    def apply_palette(self, img, palette, dither=Image.NONE):
127        """
128        Convert an RGB image to a 16-color image using a given palette.
129        
130        :param img: PIL Image in RGB mode.
131        :param palette: List of 16 RGB tuples [(R,G,B), (R,G,B), ...]
132        :param dither: Dithering mode (Image.NONE or Image.FLOYDSTEINBERG)
133        :return: PIL Image in 'P' mode using the given palette.
134        """
135        # Create a new palette image (must be a multiple of 3 in size)
136        palette_img = Image.new("P", (1, 1))
137        flat_palette = [color for rgb in palette for color in rgb]  # Flatten list
138        palette_img.putpalette(flat_palette + [0] * (768 - len(flat_palette)))  # Ensure 256 colors
139        # Convert image using the palette
140        return img.convert("RGB").quantize(palette=palette_img, dither=dither)

146    def find_colors(self, im, x, y):
147        """Find the most frequent color and the highest-contrast second frequent color in an 8x8 region."""
148        color_counts = Counter()
149        # Collect colors in the 8x8 block
150        for y8 in range(8):
151            for x8 in range(8):
152                color = im.getpixel((x + x8, y + y8))
153                color_counts[color] += 1
154
155        # Sort colors by frequency
156        sorted_colors = color_counts.most_common()
157        if not sorted_colors:
158            return None, None
159
160        most_frequent = sorted_colors[0][0]  # Most frequent color
161
162        # Find the second most frequent color with max contrast
163        max_contrast_color = None
164        max_contrast = -1
165
166        for color, _ in sorted_colors[1:]:  # Skip the most frequent color
167            contrast = self.color_distance(most_frequent, color)
168            if contrast > max_contrast:
169                max_contrast = contrast
170                max_contrast_color = color
171
172        return most_frequent, max_contrast_color
